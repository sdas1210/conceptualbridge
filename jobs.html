  • ${title} ${dateStr}
    • `; if (categoryType === "Result") { document.getElementById('results-list').insertAdjacentHTML('beforeend', itemHtml); } else if (categoryType === "Admit Card") { document.getElementById('admit-list').insertAdjacentHTML('beforeend', itemHtml); } else { document.getElementById('vacancies-list').insertAdjacentHTML('beforeend', itemHtml); } }); checkEmptyLists(); } else { throw new Error(data.message || "Unexpected server response layout."); } } catch (err) { console.error("Feed error:", err); document.querySelectorAll('.loading-text').forEach(el => { el.style.color = '#ff8080'; el.innerText = `Sync Error: ${err.message}`; }); } } function openJobModal(index, categoryType) { const targetItem = currentLoadedItems[index]; if (!targetItem) return; document.getElementById('modal-job-title').innerText = targetItem.title; document.getElementById('modal-job-date').innerText = targetItem.pubDate ? targetItem.pubDate.split(' ')[0] : 'Live'; if (categoryType === "Vacancy") { metadataRows.forEach(rowId => document.getElementById(rowId).style.display = 'table-row'); document.getElementById('modal-job-lastdate').innerText = targetItem.lastDate || "Refer to Source Document"; document.getElementById('modal-job-agelimit').innerText = targetItem.ageLimit || "18 - 38 Years"; document.getElementById('modal-job-reservation').innerText = targetItem.reservation || "APPLICABLE"; document.getElementById('modal-job-totalvacancy').innerText = targetItem.totalVacancy || "Check Source Matrix"; document.getElementById('modal-job-qualification').innerText = targetItem.qualification || "Check Qualifications"; } else { metadataRows.forEach(rowId => document.getElementById(rowId).style.display = 'none'); } document.getElementById('modal-external-link').href = targetItem.link; document.getElementById('job-modal-overlay').style.display = 'flex'; } function closeJobModal() { document.getElementById('job-modal-overlay').style.display = 'none'; } function resetBoardWorkspace() { ['vacancies', 'admit', 'results'].forEach(type => { const loader = document.getElementById(`${type}-loader`); const list = document.getElementById(`${type}-list`); list.innerHTML = ''; loader.style.color = '#cbd5e1'; loader.style.display = 'block'; loader.innerText = 'Fetching live notifications...'; }); } function switchJobFeed(targetSource) { if (activeSourceSelection === targetSource) return;
